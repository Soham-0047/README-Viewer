    Cloud Sync & Account

    Paste your Firebase config to enable cross-device sync and Google sign-in. Each user's files are private — stored at users/{uid}/files.

    Setup guide (3 minutes)
    1. Open console.firebase.google.com
    2. Create project → Add app → Web → copy firebaseConfig
    3. Build → Firestore Database → Create → test mode
    4. Build → Authentication → Sign-in method → enable Google
    5. Add your GitHub Pages domain to Authorized domains (Auth → Settings)
    6. Paste config below → Connect
    Firestore security rules (paste after connecting) 
    rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Each user can only read/write their own files
    match /users/{uid}/files/{fileId} {
      allow read, write: if request.auth != null
                         && request.auth.uid == uid;
    }
    // Share links: anyone can read, only auth users create
    match /mv_shared_links/{token} {
      allow read:  if true;
      allow write: if request.auth != null;
    }
  }
}
